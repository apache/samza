019package org.apache.hadoop.lib.wsrs;
020
021import org.apache.hadoop.classification.InterfaceAudience;
022import org.apache.hadoop.util.StringUtils;
023
024import java.util.Arrays;
025
026@InterfaceAudience.Private
027public abstract class EnumParam<E extends Enum<E>> extends Param<E> {
028  Class<E> klass;
029
030  public EnumParam(String name, Class<E> e, E defaultValue) {
031    super(name, defaultValue);
032    klass = e;
033  }
034
035  @Override
036  protected E parse(String str) throws Exception {
037    return Enum.valueOf(klass, str.toUpperCase());
038  }
039
040  @Override
041  protected String getDomain() {
042    return StringUtils.join(",", Arrays.asList(klass.getEnumConstants()));
043  }
044
045}
