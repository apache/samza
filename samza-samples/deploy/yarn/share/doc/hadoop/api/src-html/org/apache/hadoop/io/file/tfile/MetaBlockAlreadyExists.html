024

026 * Exception - Meta Block with the same name already exists.

029@InterfaceAudience.Public
030@InterfaceStability.Stable

032  /**
033   * Constructor
034   * 
035   * @param s
036   *          message.
037   */
038  MetaBlockAlreadyExists(String s) {
039    super(s);
040  }
041}
