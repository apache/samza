019package org.apache.hadoop.mapreduce.lib.db;
020
021import java.sql.Connection;
022import java.sql.ResultSet;
023import java.sql.PreparedStatement;
024import java.sql.SQLException;
025
026import org.apache.hadoop.classification.InterfaceAudience;
027import org.apache.hadoop.classification.InterfaceStability;
028import org.apache.hadoop.conf.Configuration;
029
030/**
031 * A RecordReader that reads records from a Oracle table via DataDrivenDBRecordReader
032 */
033@InterfaceAudience.Public
034@InterfaceStability.Evolving
035public class OracleDataDrivenDBRecordReader<T extends DBWritable>
036    extends DataDrivenDBRecordReader<T> {
037
038  public OracleDataDrivenDBRecordReader(DBInputFormat.DBInputSplit split,
039      Class<T> inputClass, Configuration conf, Connection conn,
040      DBConfiguration dbConfig, String cond, String [] fields,
041      String table) throws SQLException {
042
043    super(split, inputClass, conf, conn, dbConfig, cond, fields, table,
044        "ORACLE");
045
046    // Must initialize the tz used by the connection for Oracle.
047    OracleDBRecordReader.setSessionTimeZone(conf, conn);
048  }
049}
