019package org.apache.hadoop.mapred;
020
021import org.apache.hadoop.classification.InterfaceAudience;
022import org.apache.hadoop.classification.InterfaceStability;
023
024/** 
025 * Partitions the key space.
026 * 
027 * <p><code>Partitioner</code> controls the partitioning of the keys of the 
028 * intermediate map-outputs. The key (or a subset of the key) is used to derive
029 * the partition, typically by a hash function. The total number of partitions
030 * is the same as the number of reduce tasks for the job. Hence this controls
031 * which of the <code>m</code> reduce tasks the intermediate key (and hence the 
032 * record) is sent for reduction.</p>
033 * 
034 * @see Reducer
035 */
036@InterfaceAudience.Public
037@InterfaceStability.Stable
038public interface Partitioner<K2, V2> extends JobConfigurable {
039  
040  /** 
041   * Get the paritition number for a given key (hence record) given the total 
042   * number of partitions i.e. number of reduce-tasks for the job.
043   *   
044   * <p>Typically a hash function on a all or a subset of the key.</p>
045   *
046   * @param key the key to be paritioned.
047   * @param value the entry value.
048   * @param numPartitions the total number of partitions.
049   * @return the partition number for the <code>key</code>.
050   */
051  int getPartition(K2 key, V2 value, int numPartitions);
052}
