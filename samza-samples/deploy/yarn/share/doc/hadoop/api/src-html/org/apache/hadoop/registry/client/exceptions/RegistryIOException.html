019package org.apache.hadoop.registry.client.exceptions;
020
021import org.apache.hadoop.classification.InterfaceAudience;
022import org.apache.hadoop.classification.InterfaceStability;
023import org.apache.hadoop.fs.PathIOException;
024
025/**
026 * Base exception for registry operations.
027 * <p>
028 * These exceptions include the path of the failing operation wherever possible;
029 * this can be retrieved via {@link PathIOException#getPath()}.
030 */
031@InterfaceAudience.Public
032@InterfaceStability.Evolving
033public class RegistryIOException extends PathIOException {
034
035  /**
036   * Build an exception from any other Path IO Exception.
037   * This propagates the path of the original exception
038   * @param message more specific text
039   * @param cause cause
040   */
041  public RegistryIOException(String message, PathIOException cause) {
042    super(cause.getPath() != null ? cause.getPath().toString() : "",
043        message,
044        cause);
045  }
046
047  public RegistryIOException(String path, Throwable cause) {
048    super(path, cause);
049  }
050
051  public RegistryIOException(String path, String error) {
052    super(path, error);
053  }
054
055  public RegistryIOException(String path, String error, Throwable cause) {
056    super(path, error, cause);
057  }
058}
