059      cb.append(Consts.RIO_PREFIX + "ret = "+fname+".hashCode();\n");
060    }

062    @Override
063    void genSlurpBytes(CodeBuffer cb, String b, String s, String l) {
064      cb.append("{\n");
065      cb.append("int i = org.apache.hadoop.record.Utils.readVInt("+
066                b+", "+s+");\n");
067      cb.append("int z = org.apache.hadoop.record.Utils.getVIntSize(i);\n");
068      cb.append(s+" += z+i; "+l+" -= (z+i);\n");
069      cb.append("}\n");
070    }
071    
072    @Override
073    void genCompareBytes(CodeBuffer cb) {
074      cb.append("{\n");
075      cb.append("int i1 = org.apache.hadoop.record.Utils.readVInt(b1, s1);\n");
076      cb.append("int i2 = org.apache.hadoop.record.Utils.readVInt(b2, s2);\n");
077      cb.append("int z1 = org.apache.hadoop.record.Utils.getVIntSize(i1);\n");
078      cb.append("int z2 = org.apache.hadoop.record.Utils.getVIntSize(i2);\n");
079      cb.append("s1+=z1; s2+=z2; l1-=z1; l2-=z2;\n");
080      cb.append("int r1 = org.apache.hadoop.record.Utils.compareBytes(b1,s1,i1,b2,s2,i2);\n");
081      cb.append("if (r1 != 0) { return (r1<0)?-1:0; }\n");
082      cb.append("s1+=i1; s2+=i2; l1-=i1; l1-=i2;\n");
083      cb.append("}\n");
084    }
