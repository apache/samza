019package org.apache.hadoop.mapreduce.lib.join;
020
021import java.io.IOException;
022
023import org.apache.hadoop.classification.InterfaceAudience;
024import org.apache.hadoop.classification.InterfaceStability;
025import org.apache.hadoop.io.Writable;
026import org.apache.hadoop.io.WritableComparable;
027import org.apache.hadoop.mapreduce.RecordReader;
028
029/**
030 * Additional operations required of a RecordReader to participate in a join.
031 */
032@InterfaceAudience.Public
033@InterfaceStability.Stable
034public abstract class ComposableRecordReader<K extends WritableComparable<?>,
035                                             V extends Writable>
036    extends RecordReader<K,V>
037    implements Comparable<ComposableRecordReader<K,?>> {
038
039  /**
040   * Return the position in the collector this class occupies.
041   */
042  abstract int id();
043
044  /**
045   * Return the key this RecordReader would supply on a call to next(K,V)
046   */
047  abstract K key();
048
049  /**
050   * Clone the key at the head of this RecordReader into the object provided.
051   */
052  abstract void key(K key) throws IOException;
053
054  /**
055   * Create instance of key.
056   */
057  abstract K createKey();
058  
059  /**
060   * Create instance of value.
061   */
062  abstract V createValue();
063  
064  /**
065   * Returns true if the stream is not empty, but provides no guarantee that
066   * a call to next(K,V) will succeed.
067   */
068  abstract boolean hasNext();
069
070  /**
071   * Skip key-value pairs with keys less than or equal to the key provided.
072   */
073  abstract void skip(K key) throws IOException, InterruptedException;
074
075  /**
076   * While key-value pairs from this RecordReader match the given key, register
077   * them with the JoinCollector provided.
078   */
079  @SuppressWarnings("unchecked")
080  abstract void accept(CompositeRecordReader.JoinCollector jc, K key) 
081      throws IOException, InterruptedException;
082}
