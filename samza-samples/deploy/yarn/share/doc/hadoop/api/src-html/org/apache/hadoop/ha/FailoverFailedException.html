018package org.apache.hadoop.ha;
019
020import org.apache.hadoop.classification.InterfaceAudience;
021import org.apache.hadoop.classification.InterfaceStability;
022
023/**
024 * Exception thrown to indicate service failover has failed.
025 */
026@InterfaceAudience.Public
027@InterfaceStability.Evolving
028public class FailoverFailedException extends Exception {
029  private static final long serialVersionUID = 1L;
030
031  public FailoverFailedException(final String message) {
032    super(message);
033  }
034
035  public FailoverFailedException(String message, Throwable cause) {
036    super(message, cause);
037  }
038}
